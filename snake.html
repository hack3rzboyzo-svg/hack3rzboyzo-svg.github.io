<p>Arrow keys to move • R to restart</p>


<canvas id="game" width="600" height="400"></canvas>

<br>

<button onclick="resetGame()">Restart</button>


<script>

const canvas = document.getElementById("game");

const ctx = canvas.getContext("2d");


const CELL = 10;

let snake, direction, apple, gameOver, score;


function randomCell() {

    return {

        x: Math.floor(Math.random() * (canvas.width / CELL)) * CELL,

        y: Math.floor(Math.random() * (canvas.height / CELL)) * CELL

    };

}


function resetGame() {

    snake = [{x:100,y:100},{x:90,y:100},{x:80,y:100}];

    direction = {x:CELL,y:0};

    apple = randomCell();

    gameOver = false;

    score = 0;

}


document.addEventListener("keydown", e => {

    if (e.key === "ArrowUp") direction = {x:0,y:-CELL};

    if (e.key === "ArrowDown") direction = {x:0,y:CELL};

    if (e.key === "ArrowLeft") direction = {x:-CELL,y:0};

    if (e.key === "ArrowRight") direction = {x:CELL,y:0};

    if (e.key === "r") resetGame();

});


function update() {

    if (gameOver) return;


    const head = {

        x: snake[0].x + direction.x,

        y: snake[0].y + direction.y

    };


    // collision

    if (

        head.x < 0 || head.y < 0 ||

        head.x >= canvas.width || head.y >= canvas.height ||

        snake.some(s => s.x === head.x && s.y === head.y)

    ) {

        gameOver = true;

        return;

    }


    snake.unshift(head);


    if (head.x === apple.x && head.y === apple.y) {

        apple = randomCell();

        score++;

    } else {

        snake.pop();

    }

}


function draw() {

    ctx.fillStyle = "black";

    ctx.fillRect(0,0,canvas.width,canvas.height);


    // snake

    ctx.fillStyle = "lime";

    snake.forEach(s => ctx.fillRect(s.x,s.y,CELL,CELL));


    // apple

    ctx.fillStyle = "red";

    ctx.fillRect(apple.x,apple.y,CELL,CELL);


    // score

    ctx.fillStyle = "white";

    ctx.fillText("Score: " + score, 10, 15);


    if (gameOver) {

        ctx.fillStyle = "yellow";

        ctx.font = "40px monospace";

        ctx.fillText("GAME OVER", 150, 200);

    }

}


function gameLoop() {

    update();

    draw();

}


resetGame();

setInterval(gameLoop, 100);

</script>
